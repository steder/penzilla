# Let's define a class quick:
class Student:
    def __init__(self, name="", school="", major="", year="", gpa=-1):
        self.name = name
        self.school = school
        self.major = major
        self.year = year
        self.gpa = gpa

    def _Print(self):
        string = "++++++++++++++++++++++" + \
                 "+"+name \
                 "+"+school \
                 "+"+major \
                 "+"+year \
                 "+"+gpa \
                 "++++++++++++++++++++++"
        return string

    def Print(self):
        print self._Print()

    def __str__(self):
        return self._Print()
